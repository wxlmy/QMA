### 启动过程添加定时器

```bash shell
function add_cron() 
{
    item="./index_server_watch.sh"
    exist=`crontab -l | grep "$item" | grep -v "#" | wc -l`
    if [ $exist -eq 0 ]
    then
        echo "begin add crontab"
        tmpcron=`mktemp`
        crontab -l > $tmpcron
        echo "*/1 * * * * cd /data/pcdn/pcdn_index_server/bin; ./index_server_watch.sh  >> ../log/crontab.log 2>&1" >> $tmpcron
        crontab $tmpcron
        rm $tmpcron
        echo "add crontab ok"
    fi
}
```

### 停止过程删除定时器

```bash
function del_cron()
{
    item="./index_server_watch.sh"
    exist=`crontab -l | grep "$item" | grep -v "#" | wc -l`
    if [ "$exist" != "0" ]
    then
        echo "begin del crontab"
        tmpcron=`mktemp`
        crontab -l | grep -v "$item" > $tmpcron
        crontab $tmpcron
        rm $tmpcron
        echo "del crontab ok"
    fi
}
```
